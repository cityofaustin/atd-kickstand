## Install

```bash
npm install --save atd-kickstand
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'atd-kickstand'
import 'atd-kickstand/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
